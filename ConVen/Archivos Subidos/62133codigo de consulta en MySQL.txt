
--CODIGO PARA CONSULTAR INVENTARIO

SELECT `meta_value` AS CODIGO,
(SELECT A.meta_value  FROM db90970_web_db.wp_postmeta A WHERE A.post_id=B.post_id AND A.meta_key='_STOCK') AS STOCK,

(SELECT A.meta_value FROM db90970_web_db.wp_postmeta A where A.meta_key='_price' AND A.meta_value<>'' and A.post_id=B.post_id) AS PRECIO,
`post_id`,`meta_key`

FROM db90970_web_db.wp_postmeta B where   B.meta_key='_SKU' and meta_value<>''



--CODIGO PARA CONSULTAR EL PRECIO

SELECT LTRIM(B.meta_value) AS meta_value, B.post_id ,

(SELECT MAX(A.`meta_value`) FROM `wp_postmeta` A WHERE A.`post_id`=B.`post_id` AND A.`meta_key`='_SKU') AS ARTICULO

FROM db90970_web_db.wp_postmeta B where B.meta_key='_price' AND meta_value<>''
